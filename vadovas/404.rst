   <script>
     // jei kelias jau turi versiją (pvz., /latest/), nieko nedarom
     if (!/\/(latest|stable)\//.test(location.pathname)) {
       var target = '/latest' + location.pathname + location.search + location.hash;
       location.replace(target);
     }
   </script>
